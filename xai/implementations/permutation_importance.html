Permutation Feature Importance Demo

This demo showcases how to compute and visualize permutation feature importance for a given model using PyScript.

import numpy as np import matplotlib.pyplot as plt from sklearn.datasets import load_iris from sklearn.ensemble import RandomForestClassifier from sklearn.inspection import permutation_importance # Load dataset data = load_iris() X, y = data.data, data.target # Fit model model = RandomForestClassifier() model.fit(X, y) # Calculate permutation feature importance result = permutation_importance(model, X, y, n_repeats=10, random_state=42) # Sort and visualize feature importance indices = np.argsort(result.importances_mean)[::-1] features = data.feature_names # Plot plt.figure(figsize=(10, 6)) plt.title("Permutation Feature Importance") plt.barh(range(len(features)), result.importances_mean[indices], yerr=result.importances_std[indices], align="center") plt.yticks(range(len(features)), np.array(features)[indices]) plt.xlabel("Importance Score") plt.grid(True) plt.tight_layout() # Save the plot to a file plt.savefig("permutation_importance.png")
